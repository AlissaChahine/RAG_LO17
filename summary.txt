    "        writer = csv.writer(f)\n",

    "        for p in paragraphs:\n",

    "def scrape_wikipedia(title: str):\n",
    "\n",
    "    url = \"https://fr.wikipedia.org/w/api.php\"\n",
    "\n",
    "    params = {\n",
    "        \"action\": \"parse\",\n",
    "        \"page\": title,\n",
    "        \"title\": \"wikipedia\",\n",
    "        \"format\": \"json\",\n",
    "        \"prop\": \"text\",\n",
    "        \"redirects\": True\n",
    "    }\n",
    "\n",
    "    r = requests.get(url, params=params, headers={\"User-Agent\": \"Mozilla/5.0\"})\n",
    "    data = r.json()\n",
    "\n",
    "    html = data[\"parse\"][\"text\"][\"*\"]\n",
    "    soup = BeautifulSoup(html, \"html.parser\")\n",
    "\n",
    "    texts = []\n",
    "    capture = False\n",
    "\n",
    "    for tag in soup.find_all([\"h2\", \"p\"]):\n",
    "\n",
    "        t = tag.get_text(\" \", strip=True)\n",
    "\n",
    "        if tag.name == \"h2\" and \"Références\" in t:\n",
    "            break\n",
    "\n",
    "        if tag.name == \"h2\":\n",
    "            capture = True\n",
    "            continue\n",
    "\n",
    "        if capture and t:\n",
    "            texts.append(t)\n",
    "\n",
    "    write_csv(\"files/wikipedia.csv\", texts)\n",
    "\n",
    "    return Document(\n",
    "        page_content=\"\\n\\n\".join(texts),\n",
    "        metadata={\"source\": f\"wikipedia:{title}\"}\n",
    "    )"

    "import requests\n",
    "from urllib.parse import urlparse, unquote\n",
    "from langchain_core.documents import Document\n",
    "\n",
    "\n",
    "def scrape_fandom(url: str):\n",
    "\n",
    "    parsed = urlparse(url)\n",
    "\n",
    "    if \"/wiki/\" not in parsed.path:\n",
    "        return None\n",
    "\n",
    "    page_name = unquote(\n",
    "        parsed.path.split(\"/wiki/\")[-1]\n",
    "    )\n",
    "\n",
    "    parts = parsed.path.split(\"/\")\n",
    "\n",
    "    lang = parts[1] if len(parts) > 2 else \"\"\n",
    "\n",
    "    api_url = (\n",
    "        f\"{parsed.scheme}://{parsed.netloc}/{lang}/api.php\"\n",
    "    )\n",
    "\n",
    "    params = {\n",
    "        \"action\": \"query\",\n",
    "        \"prop\": \"extracts\",\n",
    "        \"titles\": page_name,\n",
    "        \"format\": \"json\",\n",
    "        \"explaintext\": True,\n",
    "        \"redirects\": 1,\n",
    "    }\n",
    "\n",
    "    response = requests.get(\n",
    "        api_url,\n",
    "        params=params,\n",
    "        headers={\"User-Agent\": \"Mozilla/5.0\"},\n",
    "        timeout=30,\n",
    "    )\n",
    "\n",
    "    data = response.json()\n",
    "\n",
    "    pages = data.get(\"query\", {}).get(\"pages\", {})\n",
    "\n",
    "    if not pages:\n",
    "        return None\n",
    "\n",
    "    page = next(iter(pages.values()))\n",
    "\n",
    "    text = page.get(\"extract\", \"\").strip()\n",
    "\n",
    "    if not text:\n",
    "        return None\n",
    "\n",
    "    return Document(\n",
    "        page_content=text,\n",
    "        metadata={\n",
    "            \"source\": url,\n",
    "            \"title\": page.get(\"title\", page_name),\n",
    "        },\n",
    "    )"
   ]

    "def enrich_with_source(docs):\n",
    "    enriched = []\n",
    "    for d in docs:\n",
    "        src = d.metadata.get(\"source\", \"unknown\")\n",
    "        source_type = \"WIKIPEDIA\" if \"wikipedia\" in src else \"FANDOM\"\n",
    "\n",
    "        # Extraire le nom de la page depuis l'URL comme \"pseudo-titre\"\n",
    "        if \"wiki/\" in src:\n",
    "            page_title = src.split(\"wiki/\")[-1].replace(\"_\", \" \")\n",
    "        elif \"wikipedia:\" in src:\n",
    "            page_title = src.split(\"wikipedia:\")[-1].replace(\"_\", \" \")\n",
    "        else:\n",
    "            page_title = src\n",
    "\n",
    "        # Enrichissement du chunk : on ajoute le contexte AVANT le texte\n",
    "        # → la similarité vectorielle bénéficie du titre + type de source\n",
    "        d.page_content = (\n",
    "            f\"SOURCE: {src}\\n\"\n",
    "            f\"SOURCE_TYPE: {source_type}\\n\"\n",
    "            f\"TOPIC: {page_title}\\n\\n\"   # ← nouveauté\n",
    "            f\"{d.page_content}\"\n",
    "        )\n",
    "        enriched.append(d)\n",
    "    return enriched\n",
    "\n",
    "split_docs = enrich_with_source(split_docs)"
